            package Paquete;

            import java.util.Scanner;
            
            
            public class Vector {
                private int[] dato = new int[5];
                private Scanner leer = new Scanner(System.in);
                
                public void Vector(){
                    for(int i = 0; i < dato.length; i++){
                    LeerNumero(i);        
                    dato[i] = capturarDatoInt();
                    }
                    for(int j = 0; j < dato.length; j++){
                    MostrarDato(j);
                    }
                   
            
                }
                
                private void LeerNumero(int i){
                    System.out.println("Ingrese numero de la posicion "+(i+1));
                    
                }
                
                private int capturarDatoInt(){
                    return leer.nextInt();
                }
                
                private void MostrarDato(int j){
                    System.out.println("["+j+"] = "+dato[j]);
                    
                }
            }

            package Paquete;



            public class Par_Impar {
                private int[] matriz = new int[20];
                
                public void Par_Impar(){
                    for (int i = 0; i < matriz.length; i++) {
                        matriz[i] = (int) (Math.random() * (100 - 1 + 1) + 1);
                    }
                    System.out.print("Numeros pares: ");
                    for(int j = 0; j < matriz.length; j++){
                    MostrarPar(j);
                    }
                    System.out.println("");
                    System.out.print("Numeros impares: ");
                    for(int j = 0; j < matriz.length; j++){
                    MostrarImpar(j);
                    }
                    System.out.println("");
                }
                
                
                private void MostrarPar(int j){
                   if (matriz[j] % 2 == 0) {
                            System.out.print(" "+matriz[j]);
                        }
                    
                }
                
                private void MostrarImpar(int j){
                    if (matriz[j] % 2 != 0) {
                            System.out.print(" "+matriz[j]);
                        }
                }
            }
