Talleres en PSeInt - Taller de Arreglos

1-Crear un vector de tipo Entero con 5 posiciones, llenarlo con información solicitada al usuario. Después de recoger toda la información, se requiere imprimir el índice de cada posición en el arreglo con su valor de la siguiente manera:

Codigo:


          Proceso Taller_04_1
            Definir contador Como Entero;
            Definir matriz Como Entero;
            Dimension matriz[5];
            
            //Leer numeros
            Para contador <- 0 Hasta 4 Con Paso 1 Hacer
              contador <- contador+1;//numero de pregunta
              Escribir "Ingrese un numero ",contador;
              contador <- contador-1;//numero de posicion para asiganar
              Leer matriz[contador];
            FinPara
            
            Limpiar Pantalla;
            Escribir "Lista de numeros:";
            
            //mostrar numeros
            Para contador <- 0 Hasta 4 Con Paso 1 Hacer
              Escribir "[",contador,"]"," = ",matriz[contador];
            FinPara
            
          FinProceso

2-Crear un arreglo de números enteros de 20 posiciones, el cual, debe ser llenado con números aleatorios entre 1 y 100; después de haber llenado dicho arreglo, se debe volver a recorrer utilizando un ciclo diferente al que se usó para llenarse e imprimir los números pares e impares. Ejemplo

Codigo:


          Proceso Taller_04_2
            Definir contador,pares,impares Como Entero;
            Definir matriz,mpares,mimpares Como Entero;
            Dimension matriz[20];
            Dimension mpares[20];
            Dimension mimpares[20];
            pares<-0;
            impares<-0;
            
            
            Para contador <- 0 Hasta 19 Con Paso 1 Hacer
              
              Repetir
                
                contador <- contador+1;
                Escribir "Escoja 20 numeros entre 1 y 100 Ingrese el numero: ",contador;
                contador <- contador-1;
                Leer matriz[contador];
                
                
              Hasta Que matriz[contador] <= 100 y matriz[contador] >= 1
              
            FinPara
            
            Limpiar Pantalla;
            
            contador <- 0;
            
            Mientras contador <= 19 Hacer
              
              Si matriz[contador]%2 = 0 Entonces
                mpares[pares] <- matriz[contador];
                
                pares <- pares+1;
              SiNo
                mimpares[impares] <- matriz[contador];
                
                impares <- impares+1;
              FinSi
              
              contador <- contador+1;
            FinMientras
            
            Escribir "Números pares: " Sin Saltar;
            contador <- 0;
            
            Mientras contador < pares Hacer
              Si contador = 0 Entonces
                Escribir mpares[contador]Sin Saltar ;
                contador <- contador+1;
              FinSi
              Escribir ",",mpares[contador]Sin Saltar ;
              contador <- contador+1;
            FinMientras
            
            Escribir " ";
            
            Escribir "Números impares: " Sin Saltar;
            contador <- 0;
            
            Mientras contador < impares Hacer
              Si contador = 0 Entonces
                Escribir mimpares[contador]Sin Saltar ;
                contador <- contador+1;
              FinSi
              Escribir ",",mimpares[contador]Sin Saltar ;
              contador <- contador+1;
            FinMientras
            Escribir "";
            
          FinProceso

          Proceso Taller_04_3

            Para contador <- 0 Hasta 999 Con Paso 1 Hacer
              
              
              matriz[contador] <- contador+1;
              
            FinPara
            
            Limpiar Pantalla;
            Escribir "Lista de numeros primos: " Sin Saltar;
            
            
            Para contador <- 0 Hasta 999 Con Paso 1 Hacer
              n<-matriz[contador];
              cont <- 0;
              Para i <- 1 Hasta n Con Paso 1 Hacer
                Si n%i=0 Entonces
                  cont<-cont+1;
                FinSi
              FinPara
              
              Si cont=2 Entonces 
                Si n = 997 Entonces
                  Escribir n;
                SiNo
                  Escribir n,",",Sin Saltar;
                FinSi
                
              FinSi 
            FinPara
            
          FinProceso
