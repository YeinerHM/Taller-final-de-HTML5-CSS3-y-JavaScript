                    do{
                       MostrarMenu();

                    switch(opcion){
                        case 1 -> {
                            capturarDatoString();
                            System.out.println("Ingrese su nombre");
                            nombre = capturarDatoString();
                        }
                        case 2 -> {
                            if(nombre==null){
                                System.out.println("====================Ingrese un nombre primero====================");
                            }else{
                                System.out.println("====================Hola "+nombre+" que tal====================");
                            }
                        }
                        case 3 -> {System.out.println("Cerrando Sistema");
                        }
                        default -> {System.out.println("La opcion Elegida no existe");
                        }
                    } 
                    }while(opcion!=3);
                }
                
                private void MostrarMenu(){
                    System.out.println("Menu de usuario");
                    System.out.println("1-Capturar nombre");
                    System.out.println("2-Saludar persona");
                    System.out.println("3-Salir del sistema");
                    
                }
                
                private int capturarDatoInt(){
                    return leer.nextInt();
                }
                private String capturarDatoString(){
                    return leer.nextLine();
                }
                
            }
