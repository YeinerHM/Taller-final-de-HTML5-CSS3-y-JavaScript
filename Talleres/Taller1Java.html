3-Realizar un programa el cual solicite su nombre y apellidos, también debe capturar nombre y apellidos de su padre y madre. Al finalizar debe imprimir en pantalla el siguiente mensaje "Yo [Nombre Completo], soy hijo de [Nombre de la Madre] y [Nombre del Padre].

        punto.Punto3();
    }

Codigo Punto3:


            package Punto3;

import java.util.Scanner;

public class Punto3 {
    
    private String nombre;
    private String apellido;
    private String nombreP;
    private String apellidoP;
    private String nombreM;
    private String apellidoM;
    private Scanner input = new Scanner (System.in);
    
    
    public void Punto3(){
        
        solicitarNombre();
        nombre = capturarDatoString();
        
        solicitarApellido();
        apellido = capturarDatoString();
        
        solicitarNombreP();
        nombreP = capturarDatoString();
        
        solicitarApellidoP();
        apellidoP = capturarDatoString();
        
        solicitarNombreM();
        nombreM = capturarDatoString();
        
        solicitarApellidoM();
        apellidoM = capturarDatoString();
        
        imprimirMensaje();
    }
    
    //Metodos
    private void solicitarNombre(){System.out.print("Por favor digite su nombre: ");}
    private void solicitarApellido(){System.out.print("Por favor digite sus apellidos: ");}
    private void solicitarNombreP(){System.out.print("Por favor dijite nombre de su padre: ");}
    private void solicitarApellidoP(){System.out.print("Por favor dijite apellidos de su padre: ");}
    private void solicitarNombreM(){System.out.print("Por favor dijite nombre de su madre: ");}
    private void solicitarApellidoM(){System.out.print("Por favor dijite apellidos de su madre: ");}
    private void imprimirMensaje(){System.out.println("Yo "+ nombre+" "+apellido+", soy hijo de "+ nombreM+" "+apellidoM+" "+"y "+ nombreP+" "+apellidoP);}
    
    private String capturarDatoString(){return input.nextLine();}
    
}
